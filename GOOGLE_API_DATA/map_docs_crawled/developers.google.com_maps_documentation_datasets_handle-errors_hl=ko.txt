"status": "UNAUTHENTICATED",
"details": [
"@type": "type.googleapis.com/google.rpc.ErrorInfo",

"metadata": {
"method": "google.maps.mapsplatformdatasets.v1alpha.MapsPlatformDatasetsV1Alpha.ListDatasets",
